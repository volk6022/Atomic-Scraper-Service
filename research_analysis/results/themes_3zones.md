### 2. ТЕХНИЧЕСКАЯ РЕАЛИЗАЦИЯ

```python
# Пример структуры валидации
class ResearchAgentValidator:
    def __init__(self):
        self.priority_order = [
            'contact_info',
            'legal_data', 
            'social_media',
            'scale_indicators',
            'deep_dive',
            'vacancies',
            'verification'
        ]
    
    def validate_data(self, data):
        for priority in self.priority_order:
            if priority not in data:
                data[priority] = self.get_default(priority)
        
        return self.verify_sources(data)
    
    def get_default(self, priority):
        defaults = {
            'contact_info': {'email': '', 'phone': '', 'website': ''},
            'legal_data': {'inn': '', 'ogrn': '', 'year': ''},
            'social_media': {'instagram': '', 'vk': '', 'telegram': ''},
            'scale_indicators': {'branches': 0, 'employees': 0},
            'deep_dive': {},
            'vacancies': [],
            'verification': {'sources': []}
        }
        return defaults.get(priority, {})
    
    def verify_sources(self, data):
        # Проверка на пустые sources
        if not data.get('verification', {}).get('sources'):
            data['verification']['sources'] = []
        
        # Проверка на галлюцинации
        if data.get('legal_data', {}).get('inn') and not self.is_valid_inn(data['legal_data']['inn']):
            data['legal_data']['inn'] = None
        
        return data
```
